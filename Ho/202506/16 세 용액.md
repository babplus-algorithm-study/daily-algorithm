import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.StringTokenizer;

// 10:32
public class Main {
    static int N;
    static int[] arr,ans;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());
        arr = new int[N];

        StringTokenizer st = new StringTokenizer(br.readLine());

        for (int i = 0; i < N; i++) {
            arr[i] = Integer.parseInt(st.nextToken());
        }

        Arrays.sort(arr);

        /**
         * 풀이 방법 2개 고정하고 이분탐색
         * 두개 고정하고 하나는 이분탐색으로 찾으면서 가장 작은 거 찾기
         */
        ans = new int[3];
        int[] realAns = new int[3];
        long minAns = Long.MIN_VALUE;

        for (int i = 0; i < N; i++) {
            ans[0] = arr[i];
            for (int j = i + 1; j < N; j++) {
                ans[1] = arr[j];

                //i, j 조합에 가장 어울리는 남은 하나 이분탐색으로 찾기
                long result = 0 - Math.abs(binarySearch(j + 1, N-1));
                if(minAns < result) {
                    for (int k = 0; k < 3; k++) {
                        realAns[k] = ans[k];
                    }
                    minAns = result;
                }
            }
        }
        for (int i = 0; i < 3; i++) {
            System.out.print(realAns[i] + " ");
        }

    }

    private static long binarySearch(int left, int right) {
        int mid = (left + right) / 2;
        int temp = ans[0] + ans[1];

        ans[2] = arr[mid];
        long curMin = 0 - Math.abs(ans[0] + ans[1] + arr[mid]);

        while(left <= right) {
            mid = (left + right) / 2;
            int curNum = arr[mid];

            if(curMin < 0 - Math.abs(temp + curNum)) {
                //0에 가까워 지는 경우
                ans[2] = arr[mid];
                left = mid + 1;
                curMin = 0 - Math.abs(temp + curNum);
            }
            else{
                right = mid - 1;
            }
        }
        return curMin;
    }
}
