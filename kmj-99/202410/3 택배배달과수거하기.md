class Solution {
    fun solution(cap: Int, n: Int, deliveries: IntArray, pickups: IntArray): Long {
        var sumDeliveries = 0
        var sumPickups = 0
        var answer: Long = 0
        
        for(i in n-1 downTo 0){
            var count = 0
            sumDeliveries += deliveries[i] 
            sumPickups += pickups[i]
            
            while(sumDeliveries>0 || sumPickups>0){
                sumDeliveries -= cap
                sumPickups -= cap
                count+=1
            }
    
            answer += (i+1)*count*2
        }
        
        return answer
    }
}
